226 public static bool CanFitCapacity(double[] capacity, Aggregation[] aggregationType, IEnumerable<double[]> sizes)
227 {
228 for (int i = 0; i < capacity.Length; i++)
229 {
230 if (aggregationType[i] == Aggregation.Sum)
231 {
232 if (capacity[i] < sizes.Sum(r => r[i]))
233 {
234 return false;
235 }
236 }
237 else if (aggregationType[i] == Aggregation.Max)
238 {
239 if (capacity[i] < sizes.Max(r => r[i]))
240 {
241 return false;
242 }
243 }
244 else
245 {
246 throw new ArgumentException($"Vehicle does not support {aggregationType[i]} aggregation of sizes");
247 }
248 }
249 return true;
250 }

317 public bool CanHandleRequest(TransportRequest candidateRequest)
318 {
319 if (this.Capacity.Length != candidateRequest.Size.Length)
320 {
321 throw new ArgumentException("Capacity properties of the vehicle do not match request size");
322 }
323 if (!candidateRequest.MaxVehicleSize.DoesVehicleFitIntoRestrictions(this.RoadProperties))
324 {
325 return false;
326 }
327 for (int i = 0; i < this.Capacity.Length; i++)
328 {
329 if (this.Capacity[i] < candidateRequest.Size[i])
330 {
331 return false;
332 }
333 }
334 for (int i = 0; i < candidateRequest.NecessaryVehicleSpecialProperties.Length; i++)
335 {
336 if (!this.SpecialProperties.Contains(candidateRequest.NecessaryVehicleSpecialProperties[i]))
337 {
338 return false;
339 }
340 }
341 return true;
