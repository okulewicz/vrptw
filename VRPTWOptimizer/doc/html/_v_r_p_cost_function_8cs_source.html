181 public static double ComputeFillInFactor(List<double[]> cargoOnRouteStart, double[] vehicleCapacity, Enums.Aggregation[] aggregationType)
182 {
183 double maxFillIn = 0.0;
184 for (int i = 0; i < vehicleCapacity.Length; i++)
185 {
186 double cargoSumI = 0.0;
187 if (aggregationType[i] == Enums.Aggregation.Sum)
188 {
189 cargoSumI = cargoOnRouteStart.Sum(c => c[i]);
190 }
191 else if (aggregationType[i] == Enums.Aggregation.Max)
192 {
193 if (cargoOnRouteStart.Any())
194 {
195 cargoSumI = cargoOnRouteStart.Max(c => c[i]);
196 }
197 }
198 maxFillIn = Math.Max(maxFillIn, cargoSumI / vehicleCapacity[i]);
199 }
200 return maxFillIn;
201 }

324 public double SingleRouteValue(IRoute route)
325 {
326 double length = route.Length;
327 double travelTime = route.TravelTime;
328 Vehicle vehicle = route.Vehicle;
329 Vehicle vehicleTractor = route.VehicleTractor;
330 IEnumerable<double> revenueValues = route.LoadedRequests.SelectMany(rq => rq.Select(rq => rq.RevenueValue));
331 double[] timeWindowsStart = route.TimeWindowStart
332 .Select(tws => tws).ToArray();
333 double[] arrivalTimes = route.ArrivalTimes
334 .Select(art => art).ToArray();
335 double vehicleDistanceCost = vehicle.ComputeDistanceCost(length);
336 double vehicleTimeCost = vehicle.ComputeTimeCost(travelTime);
337 if (vehicleTractor != null)
338 {
339 vehicleDistanceCost += route.VehicleTractor.ComputeDistanceCost(length);
340 vehicleTimeCost += route.VehicleTractor.ComputeTimeCost(travelTime);
341 }
342 //cost of being too early
343 double routeTotalEarly = ComputeTotalTimeDiff(timeWindowsStart, arrivalTimes, true);
344 double earlyArrivalsCost = TotalEarlyArrivalFactor * routeTotalEarly + TotalEarlyArrivalSquaredFactor * routeTotalEarly * routeTotalEarly;
345 //cost of delay
346 double routeTotalDelay = route.TotalDelay;
347 double lateArrivalsCost = TotalDelayFactor * routeTotalDelay + TotalDelaySquaredFactor * routeTotalDelay * routeTotalDelay;
348 return
349 DistanceFactor * vehicleDistanceCost +
350 //cost of cargo cooling
351 DriveTimeFactor * vehicleTimeCost +
352 //cost of using vehicle at all
353 lateArrivalsCost +
354 earlyArrivalsCost -
355 //gain of serving profit generating requests
356 revenueValues.Sum(val => val);
357 ;
358 }

366 public double Value(List<IRoute> routes, List<TransportRequest> leftRequests)
367 {
368 /*
369 * double vehicleUsageCost = route.Vehicle.VehicleCostPerUsage;
370 vehicleUsageCost += route.VehicleTractor.VehicleCostPerUsage;
371
372 */
373 var orderedRoutes = routes.OrderBy(rt => rt.VehicleTractor != null ? rt.VehicleTractor.Id : rt.Vehicle.Id)
374 .ThenBy(rt => rt.ArrivalTimes[0]);
375 IRoute previousRoute = null;
376 double maxVehicleSpread = 0.0;
377 foreach (var currentRoute in orderedRoutes)
378 {
379 if (previousRoute != null)
380 {
381 if (currentRoute.VehicleTractor != null)
382 {
383 if (previousRoute.VehicleTractor != null && currentRoute.VehicleTractor.Id == previousRoute.VehicleTractor.Id)
384 {
385 maxVehicleSpread = Math.Max(maxVehicleSpread, currentRoute.DepartureTimes[0] - previousRoute.ArrivalTimes[^1]);
386 }
387 }
388 else if (currentRoute.Vehicle.Id == previousRoute.Vehicle.Id)
389 {
390 maxVehicleSpread = Math.Max(maxVehicleSpread, currentRoute.ArrivalTimes[0] - previousRoute.DepartureTimes[^1]);
391 }
392 }
393 previousRoute = currentRoute;
394 }
395 double maxVehicleSpreadCost = MaxVehicleSpreadFactor * maxVehicleSpread;
396
397 var realDistanceDict = routes
398 .GroupBy(rt => rt.Vehicle.OwnerID)
399 .ToDictionary(gr => gr.Key, gr => gr.Sum(rt => rt.Length));
400 var totalDistance = routes.Sum(rt => rt.Length);
401 var lowFillInCost = routes.Any() ? FillInFactor * routes.Average(route => (1 - ComputeFillInFactor(route))) : 0.0;
402 var maxEarlyArrivalCost = routes.Any() ? routes.Max(route =>
403 {
404 double routeMaxEarlyArrival = ComputeMaxEarlyArrival(route);
405 return MaxEarlyArrivalFactor * routeMaxEarlyArrival + MaxEarlyArrivalSquaredFactor * routeMaxEarlyArrival * routeMaxEarlyArrival;
406 }) : 0.0;
407 var maxLaterArrivalCost = routes.Any() ? routes.Max(route =>
408 {
409 double maxDelay = route.MaxDelay;
410 return MaxDelayFactor * maxDelay + MaxDelaySquaredFactor * maxDelay * maxDelay;
411 }) : 0.0;
412 var routesCost = routes.Sum(
413 //cost of routes
414 rt => SingleRouteValue(rt));
415 //cost of not delivering certain amount of cargo
416 var leftCargoCost = LeftCargoUnitFactor * leftRequests.Sum(lr => lr.PackageCount);
417 var carrierCost = //cost of not getting equal routes distance share between car owners
418 routes.Any() ? CarrierShareFactor * CarrierShareRatio.Keys.Sum(csrDictKey =>
419 Math.Abs(totalDistance * CarrierShareRatio[csrDictKey] - realDistanceDict[csrDictKey]) / 2.0) +
420 //cost of not providing enough kilometers per car owner
421 CarrierMinDistanceFactor * CarrierMinDistanceThreshold.Keys.Sum(cmdtDictKey =>
422 Math.Abs(Math.Max(0, CarrierMinDistanceThreshold[cmdtDictKey] - realDistanceDict[cmdtDictKey]))) : 0.0;
423 double routeCountCost = RoutesCountFactor * routes.Count;
424 return routesCost +
425 lowFillInCost +
426 routeCountCost +
427 maxVehicleSpreadCost +
428 maxEarlyArrivalCost +
429 maxLaterArrivalCost +
430 carrierCost +
431 leftCargoCost;
432 }
