42 public VRPResultStatistics(VRPOptimizerResult newResult)
43 {
44 var newRoutes = newResult.Routes;
45 LeftRequests = newResult.LeftRequests.Count;
46 TotalTravelTime = TimeSpan.FromSeconds(newRoutes.Sum(rt => rt.Distances.Sum(dt => dt.Time)));
47 MaxDelayTime = TimeSpan.FromSeconds(newRoutes.Any() ? newRoutes.Max(rt => rt.MaxDelay) : 0);
48 MaxEarlyArrival = TimeSpan.FromSeconds(newRoutes.Any() ? newRoutes.Max(rt => VRPCostFunction.ComputeMaxEarlyArrival(rt)) : 0);
49 TotalDelay = TimeSpan.FromSeconds(newRoutes.Sum(rt => rt.TotalDelay));
50 TotalEarlyArrival = TimeSpan.FromSeconds(newRoutes.Sum(rt => VRPCostFunction.ComputeTotalEarlyArrival(rt)));
51 Routes = newRoutes.Count();
52 Length = Math.Round(newRoutes.Sum(rt => rt.Length) / 1000);
53 Cost = Math.Round(newRoutes.Sum(rt => (rt.Length < rt.Vehicle.VehicleMaxRouteLengthForFlatCost ? rt.Vehicle.VehicleFlatCostForShortRouteLength : rt.Vehicle.VehicleCostPerDistanceUnit * rt.Length)));
54 TravelTimeStr = $"{TotalTravelTime.Days * 24 + TotalTravelTime.Hours}:{TotalTravelTime.Minutes:00}";
55 FillRatio = Math.Round(newRoutes.Any() ? newRoutes.Average(rt => VRPCostFunction.ComputeFillInFactor(rt)) : 0, 2);
56 MaxDelayStr = $"{MaxDelayTime.Days * 24 + MaxDelayTime.Hours}:{MaxDelayTime.Minutes:00}";
57 TotalDelayStr = $"{TotalDelay.Days * 24 + TotalDelay.Hours}:{TotalDelay.Minutes:00}";

62 UniqueTrailersCount = newResult.Routes.Where(rt => rt.Vehicle.Type == CommonGIS.Enums.VehicleType.SemiTrailer).Select(rt => rt.Vehicle.Id).Distinct().Count();
63 UniqueTrucksCount = newResult.Routes.Where(rt => rt.Vehicle.Type == CommonGIS.Enums.VehicleType.StraightTruck).Select(rt => rt.Vehicle.Id).Distinct().Count();
64 UniqueTractorCount = newResult.Routes.Where(rt => rt.VehicleTractor != null).Select(rt => rt.VehicleTractor.Id).Distinct().Count();

143 MaxDelayTime.TotalSeconds,
144 late));

156 MaxDelayTime.TotalSeconds,

169 }
