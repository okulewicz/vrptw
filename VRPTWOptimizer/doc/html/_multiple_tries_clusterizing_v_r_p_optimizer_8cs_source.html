109 public VRPOptimizerResult Optimize(IVRPProvider problemDataProvider, VRPCostFunction costFunctionFactors, ITimeEstimator serviceTimeEstimator, IDistanceProvider distanceProvider)
110 {
111 if (problemDataProvider.Vehicles.All(v => v.Type == CommonGIS.Enums.VehicleType.StraightTruck) && problemDataProvider.Requests.All(rq => rq.PickupLocation.Id == problemDataProvider.HomeDepot.Id))
112 {
113 InnerOptimizerFactory = new DistributionClusteringIntegratedFleet.ClusterizingVRPOptimizerFactory();
114 }
115 else
116 {
117 InnerOptimizerFactory = new GeneralizedClustering.ClusterizingVRPOptimizerFactory();
118 }
119 var logger = new OptimizerLogger($"vrp-{DateTime.UtcNow.ToFileTimeUtc()}-{this.GetType()}-{Assembly.GetAssembly(this.GetType()).GetName().Version}.log");
120 InitializeDistanceProvider(problemDataProvider, distanceProvider);
121 VRPOptimizerResult globalBestResult = null;
122 double currentDelay = InitialAllowedDelay;
123 double delayStep = 600;
124 double maxDelay = MaxAllowedDelay;
125 do
126 {
127 var greedySettings = new Dictionary<string, object>();
128 greedySettings.Add(ClusterizingVRPOptimizerFactoryBase.MaxDelaySettingsKey, currentDelay);
129 greedySettings.Add(ClusterizingVRPOptimizerFactoryBase.RandomSeedSettingsKey, -1);
130 greedySettings.Add(ClusterizingVRPOptimizerFactoryBase.RandomVehiclesSettingsKey, false);
131 greedySettings.Add(ClusterizingVRPOptimizerFactoryBase.MaxVehicleDelaySettingsKey, 0);
132 var greedyCluster = InnerOptimizerFactory.CreateOptimizer(greedySettings);
133 globalBestResult = greedyCluster.Optimize(problemDataProvider, costFunctionFactors, serviceTimeEstimator, distanceProvider);
134 currentDelay += delayStep;
135 } while (globalBestResult.LeftRequests.Count > 0 && currentDelay < maxDelay);
136 double globalBestCost = costFunctionFactors.Value(globalBestResult.Routes, globalBestResult.LeftRequests);
137 int p = -1;
138 LogStats(p, globalBestResult, globalBestCost, true, logger);
139 var list = new List<int>();
140 var procs = new int[Math.Min(Environment.ProcessorCount, MaxProcessorsCount)];
141 var iters = new int[procs.Length];
142 for (int i = 0; i < procs.Length; i++)
143 {
144 procs[i] = i;
145 }
146 var results = new VRPOptimizerResult[procs.Length];
147 var computStart = DateTime.UtcNow;
148 procs.AsParallel().ForAll(p =>
149 {
150 var optimizerSettings = new Dictionary<string, object>();
151 optimizerSettings.Add(ClusterizingVRPOptimizerFactoryBase.MaxDelaySettingsKey, currentDelay);
152 optimizerSettings.Add(ClusterizingVRPOptimizerFactoryBase.RandomSeedSettingsKey, p);
153 optimizerSettings.Add(ClusterizingVRPOptimizerFactoryBase.RandomVehiclesSettingsKey, true);
154 var randomCluster = InnerOptimizerFactory.CreateOptimizer(optimizerSettings);
155 var bestResult = globalBestResult;
156 var bestCost = globalBestCost;
157
158 for (int i = 0; i < MaxIterations / procs.Length && (DateTime.UtcNow - computStart).TotalSeconds < MaxComputatationSeconds; i++)
159 {
160 var candidateResult = randomCluster.Optimize(problemDataProvider, costFunctionFactors, serviceTimeEstimator, distanceProvider);
161 var candidateCost = costFunctionFactors.Value(candidateResult.Routes, candidateResult.LeftRequests);
162
163 if (candidateCost < bestCost)
164 {
165 bestResult = candidateResult;
166 bestCost = candidateCost;
167 LogStats(p, bestResult, bestCost, false, logger);
168 }
169 iters[p] = i + 1;
170 }
171 results[p] = bestResult;
172 });
173 LogProblemParams(costFunctionFactors, serviceTimeEstimator, logger);
174 var bestSolution = results
175 .Select(rs => new KeyValuePair<VRPOptimizerResult, double>(rs, costFunctionFactors.Value(rs.Routes, rs.LeftRequests)))
176 .OrderBy(rspair => rspair.Value)
177 .First();
178 Console.WriteLine($"Total iterations: {iters.Sum()}");
179
180 LogStats(p, bestSolution.Key, bestSolution.Value, true, logger);
181 return bestSolution.Key;
182 }
