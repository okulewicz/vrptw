198 public static List<Distance> GetRouteDistances(ALNSRoute route, IVRPJSONProvider vrpProvider, IDistanceProvider distanceProvider)
199 {
200 List<Distance> Distances = new List<Distance>();
201
202 Vehicle vehicle = ALNSVehicle2Vehicle(route.Vehicle, vrpProvider);
203 VehicleRoadRestrictionProperties Profile = vehicle.RoadProperties;
204
205 for (int i = 0; i < route.Nodes.Count - 1; i++)
206 {
207 Distance distance = null;
208 if (i == 0)
209 {
210 TransportRequest toReq = Node2Request(route.Nodes[i + 1], vrpProvider);
211 Location toId = toReq.DeliveryLocation;
212 distance = distanceProvider.GetDistance(vrpProvider.HomeDepot, toId, Profile);
213 }
214 else if (i >= 1 && i < route.Nodes.Count - 2)
215 {
216 TransportRequest fromReq = Node2Request(route.Nodes[i], vrpProvider);
217 TransportRequest toReq = Node2Request(route.Nodes[i + 1], vrpProvider);
218 Location fromId = fromReq.DeliveryLocation;
219 Location toId = toReq.DeliveryLocation;
220 distance = distanceProvider.GetDistance(fromId, toId, Profile);
221
222 }
223 else
224 {
225 TransportRequest fromReq = Node2Request(route.Nodes[i], vrpProvider);
226 Location fromId = fromReq.DeliveryLocation;
227 distance = distanceProvider.GetDistance(fromId, vrpProvider.HomeDepot, Profile);
228 }
229
230 Distances.Add(distance);
231 }
232
233 return Distances;
