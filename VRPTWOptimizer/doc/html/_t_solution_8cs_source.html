85 private double RouteCost(List<TNode> nodes, List<VVehicle> vehicles_SA, List<int> r, int k, IDistanceProvider distanceProvider)
86 {
87 double routeLength = RouteLength(nodes, vehicles_SA, r, k, distanceProvider);
88 double routeTime = RouteTime(nodes, vehicles_SA, r, k, distanceProvider);
89 double cost = 0.0;
90 if (routeLength < vehicles_SA[k].vehicleFlatCostThreshold)
91 {
92 cost += vehicles_SA[k].vehicleCostFlatRate;
93 }
94 else
95 {
96 cost += routeLength * vehicles_SA[k].vehicleCostPerDistanceUnit;
97 }
98 cost += vehicles_SA[k].vehicleCostPerTimeUnit * routeTime;
99 cost += vehicles_SA[k].vehicleCostPerUsage;
100 return cost;
101 }

857 public double PrefWindowPenaltyFunction(double delta_early, double delta_late, byte penalty_type)
858 {
859 double result = 0.0;
860 switch (penalty_type)
861 {
862 case 1: // klasyczne karanie kwadratowe (z obydwu stron)
863 {
864 if (delta_early != 0)
865 {
866 result = delta_early * delta_early;
867 }
868 else
869 {
870 result = delta_late * delta_late;
871 }
872 result = result * SimmulatedAnnealingVRPTWOptimizer.PENALTYFACTORPREFTW;
873 }
874 break;
875
876 case 2: // karanie liniowe za early arrival i karanie kwadratowe za late arrival
877 {
878 if (delta_early != 0)
879 {
880 result = delta_early * SimmulatedAnnealingVRPTWOptimizer.PENALTYFACTORPREFTWLINEAR;
881 }
882 else
883 {
884 result = delta_late * delta_late * SimmulatedAnnealingVRPTWOptimizer.PENALTYFACTORPREFTW;
885 }
886 }
887 break;
888
889 case 3: // karanie kwadratowe za early arrival i karanie liniowe za late arrival
890 {
891 if (delta_early != 0)
892 {
893 result = delta_early * delta_early * SimmulatedAnnealingVRPTWOptimizer.PENALTYFACTORPREFTW;
894 }
895 else
896 {
897 result = delta_late * SimmulatedAnnealingVRPTWOptimizer.PENALTYFACTORPREFTWLINEAR;
898 }
899 }
900 break;
901 }
902 return result;
