62 VehicleRoadRestrictionProperties.MaxWidth, epCount, isTrailer ? CommonGIS.Enums.VehicleTypeRouting.TractorWithTrailer : CommonGIS.Enums.VehicleTypeRouting.StraightTruck),
63 isTrailer ? CommonGIS.Enums.VehicleType.SemiTrailer : CommonGIS.Enums.VehicleType.StraightTruck,

82 public static bool FixRequestSize(bool isFixed, List<ValidationError> errors)
83 {
84 foreach (var error in errors.Where(e => e.ErrorCode == ErrorCode.ExceededCapacity && e.ObjectType == VRPObjectType.Route))
85 {
86 isFixed = true;
87 Dictionary<TransportRequest, double[]> capacityToBeSubtracted = new();
88 var route = (error.ConsideredObject as IRoute);
89 List<VRPTWOptimizer.TransportRequest> currentStack = new();
90 for (int iloc = 0; iloc < route.VisitedLocations.Count; iloc++)
91 {
92 route.UnloadedRequests[iloc].ForEach(rq => { currentStack.Remove(rq); });
93 currentStack.AddRange(route.LoadedRequests[iloc]);
94 foreach (var loadedRequest in route.LoadedRequests[iloc])
95 {
96 capacityToBeSubtracted.Add(loadedRequest, new double[route.Vehicle.Capacity.Length]);
97 }
98 for (int i = 0; i < route.Vehicle.Capacity.Length; i++)
99 {
100 double capacitySumDim = currentStack.Sum(rq => rq.Size[i]);
101 int currentStackSize = currentStack.Count;
102 if (capacitySumDim > route.Vehicle.Capacity[i])
103 {
104 double overhead = capacitySumDim - route.Vehicle.Capacity[i];
105 foreach (var stackRequest in currentStack)
106 {
107 capacityToBeSubtracted[stackRequest][i] = Math.Max(capacityToBeSubtracted[stackRequest][i], overhead * stackRequest.Size[i] / capacitySumDim);
108 }
109 }
110 }
111 }
112 foreach (var requestToBeSubtracted in capacityToBeSubtracted)
113 {
114 for (int i = 0; i < requestToBeSubtracted.Value.Length; i++)
115 {
116 if (requestToBeSubtracted.Value[i] > 0)
117 {
118 requestToBeSubtracted.Key.Size[i] -= requestToBeSubtracted.Value[i];
119 requestToBeSubtracted.Key.Size[i] = Math.Floor(requestToBeSubtracted.Key.Size[i] * 4.0) / 4.0;
120 }
121 }
122 }
123 }
124
125 return isFixed;
126 }
