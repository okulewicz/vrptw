213 public List<TransportRequest> ExtractBestFitRequests(List<TransportRequest> requests, ITimeEstimator timeEstimator, IDistanceProvider distanceProvider)
214 {
215 VehicleRoadRestrictionProperties defaultVehicleProperties = new VehicleRoadRestrictionProperties(
216 VehicleRoadRestrictionProperties.MaxGrossVehicleWeight, 0, 0, 0, VehicleTypeRouting.TractorWithTrailer);
217 var bestFitRequests = new List<TransportRequest>();
218 double bestbackhaulingDistance = double.PositiveInfinity;
219 foreach (TransportRequest request in requests)
220 {
221 if (this.Type == RequestType.Backhauling && request.Type == RequestType.GoodsDistribution)
222 {
223 Distance distanceBetween = distanceProvider
224 .GetDistance(request.DeliveryLocation, this.PickupLocation, defaultVehicleProperties);
225 Distance backhaulingDistance = distanceProvider
226 .GetDistance(this.PickupLocation, this.DeliveryLocation, defaultVehicleProperties);
227 double unloadOldGoodsServiceTime = timeEstimator.EstimateLoadUnloadTime(request.PackageCount, 0, 0, 1);
228 double loadNewGoodsServiceTime = timeEstimator.EstimateLoadUnloadTime(0, this.PackageCount, 0, 1);
229 double arrivalTime = request.DeliveryPreferedTimeWindowStart + unloadOldGoodsServiceTime + distanceBetween.Time + loadNewGoodsServiceTime + backhaulingDistance.Time;
230 if (arrivalTime < this.DeliveryPreferedTimeWindowEnd && request.MaxVehicleSize.DoesVehicleFitIntoRestrictions(this.MaxVehicleSize) && this.MaxVehicleSize.DoesVehicleFitIntoRestrictions(request.MaxVehicleSize))
231 {
232 if (distanceBetween.Length < bestbackhaulingDistance)
233 {
234 bestFitRequests.Clear();
235 bestbackhaulingDistance = distanceBetween.Length;
236 bestFitRequests.Add(request);
237 }
238 }
239 }
240 else if (this.Type == RequestType.GoodsDistribution && request.Type == RequestType.ContainerRetrieval)
241 {
242 Distance distance = distanceProvider
243 .GetDistance(this.DeliveryLocation, request.PickupLocation, defaultVehicleProperties);
244 if (distance.Length == 0)
245 {
246 if (this.DeliveryPreferedTimeWindowStart <= request.PickupPreferedTimeWindowEnd)
247 {
248 bestFitRequests.Add(request);
249 }
250 }
251 }
252 }
253 return bestFitRequests;
254 }
