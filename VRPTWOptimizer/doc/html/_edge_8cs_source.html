25 private static void TryCreateEdge(IDistanceProvider distanceProvider, VehicleRoadRestrictionProperties maxVehicleRoadProperties, double maxDelayWithinRoute, VRPCostFunction function, List<Edge> edges, TransportRequest request1, TransportRequest request2, Location locationTo, Location locationFrom, double serviceTimeAtLocationFrom, double timeWindowStartAtLocationFrom, double timeWindowStartAtLocationTo, double timeWindowsEndAtLocationTo)
26 {
27 var distance = distanceProvider.GetDistance(locationFrom, locationTo, maxVehicleRoadProperties);
28 var waitingTimeAtRequest2 = Math.Max(timeWindowStartAtLocationTo - (timeWindowStartAtLocationFrom + serviceTimeAtLocationFrom + distance.Time), 0);
29 var lateTimeAtRequest2 = Math.Max(timeWindowStartAtLocationFrom + serviceTimeAtLocationFrom + distance.Time - timeWindowsEndAtLocationTo, 0);
30 if (lateTimeAtRequest2 <= maxDelayWithinRoute)
31 {
32 var edgeCost = function.DistanceFactor * distance.Length
33 + function.DriveTimeFactor * distance.Time
34 + Math.Max(function.MaxEarlyArrivalFactor, function.TotalEarlyArrivalFactor) * waitingTimeAtRequest2
35 + Math.Max(function.MaxDelaySquaredFactor, function.TotalDelaySquaredFactor) * lateTimeAtRequest2 * lateTimeAtRequest2;
36 edges.Add(new Edge(request1, request2, edgeCost));
37 }
