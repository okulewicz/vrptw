145 public static VRPSolution GenerateVRPSolution(
146 IVRPOptimizer _optimizer,
147 DateTime computationsStart,
148 DateTime computationsEnd,
149 List<TransportRequest> leftRequests,
150 List<IRoute> routes)
151 {
152 List<VRPSolution.TransportItem> transportItems = new();
153 var orderedTrailerAssignment = routes
154 .OrderBy(rt => rt.ArrivalTimes[0]);
155 int transportId = 1;
156 foreach (var assignment in orderedTrailerAssignment)
157 {
158 List<VRPSolution.ScheduleItem> scheduleItems = new();
159 for (int i = 0; i < assignment.VisitedLocations.Count; i++)
160 {
161 VRPSolution.ScheduleItem scheduleItem = new()
162 {
163 LocationId = assignment.VisitedLocations[i].Id,
164 ArrivalTime = assignment.ArrivalTimes[i],
165 DepartureTime = assignment.DepartureTimes[i],
166 Delay = Math.Max(assignment.ArrivalTimes[i] - assignment.TimeWindowEnd[i], 0),
167 LoadedRequestsIds = assignment.LoadedRequests[i].Select(rq => rq.Id).ToList(),
168 UnloadedRequestsIds = assignment.UnloadedRequests[i].Select(rq => rq.Id).ToList()
169 };
170 scheduleItems.Add(scheduleItem);
171 }
172 double fillInRatio = assignment.Vehicle.Capacity
173 .Select((capacity, index) => index)
174 .Max(index => assignment.LoadedRequests[0].Sum(rq => rq.Size[index]) / assignment.Vehicle.Capacity[index])
175 ;
176 VRPSolution.TransportItem transport = new()
177 {
178 TransportId = transportId,
179 TractorId = assignment.VehicleTractor == null ? -1 : assignment.VehicleTractor.Id,
180 TrailerTruckId = assignment.Vehicle.Id,
181 DriverId = assignment.VehicleDriver == null ? -1 : assignment.VehicleDriver.Id,
182 Length = assignment.Length,
183 Schedule = scheduleItems,
184 AvailableForLoadingTime = assignment.ArrivalTimes[0],
185 AvailableForNextAssignmentTime = assignment.DepartureTimes[^1],
186 FillInRatio = Math.Round(fillInRatio, 2)
187 };
188 transportItems.Add(transport);
189 transportId++;
190 }
191
192 VRPSolution vrpSolution = new()
193 {
194 LeftRequestsIds = leftRequests.Select(req => req.Id).ToList(),
195 ComputationTimestamp = computationsEnd,
196 ComputationTime = (computationsEnd - computationsStart).TotalSeconds,
197 ComputerId = Environment.MachineName,
198 Version = Assembly.GetAssembly(_optimizer.GetType()).GetName().Version,
199 DelaysCount = routes.Count(ta => ta.TotalDelay >= 0),
200 MaxDelay = routes.Any() ? routes.Max(ta => ta.MaxDelay) : 0,
201 TotalDelay = routes.Sum(ta => ta.TotalDelay),
202 TotalLength = routes.Sum(ta => ta.Length),
203 Transports = transportItems,
204 Algorithm = _optimizer.GetType().FullName
205 };
206 return vrpSolution;
207 }
208 }
